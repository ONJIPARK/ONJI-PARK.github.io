```python
import requests

    # 제목, 날짜 출력
```

# 과제2(단톡) 60191315 박온지

- https://movie.naver.com/movie/bi/mi/basic.naver?code=42809 : 영화 마음이...

- 영화 리뷰 사이트(네이버 제외)에서 자신이 본 영화 중 하나를 골라서 평점과 리뷰내용을 1페이지에서 10페이지까지 크롤링 하시오.

- 페이지 주소를 자동 생성하시오.

- 평점도 크롤링하시오.

- 데이터 프레임으로 만드시오.

- 점수와 리뷰 내용을 칼럼으로 가질 것

- 점수가 높은 리뷰부터 낮은 리뷰 순으로 정렬할 것



```python
import requests
from bs4 import BeautifulSoup
import pandas as pd

url_pre = 'https://movie.naver.com/movie/bi/mi/pointWriteFormList.naver?code=42809&type=after&isActualPointWriteExecute=false&isMileageSubscriptionAlready=false&isMileageSubscriptionReject=false&page=1'
id_list = []
id_pre = '_filtered_ment_'
df = []
        
# 10페이지 가져오기
for pg in range(10): 
     
    print('\n',pg+1,"page\n")
    site = url_pre+str(pg+1)  
    res = requests.get(site)
    soup = BeautifulSoup(res.content,'html.parser')
    
    # 점수 10개 가져오기
    grade_list = []
    grade = soup.find_all('div','star_score')
    for g in grade:
        grade_list.append(g.get_text())
    
    # 리뷰 10개 가져오기
    for i in range(10): 
        id_list.append(id_pre+str(i))

    mydata = []    
    for list in id_list:    
        mydata.append(soup.find('span',{'id':list}).get_text())
        
    # 점수, 리뷰 출력
    for gr, line in zip(grade_list, mydata): 
        print(gr.strip(), line.strip())  # -> 출력문
        df.append([gr.strip(), line.strip()])
```
